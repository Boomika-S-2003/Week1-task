üìù Variables

Variables are containers for storing data values. JavaScript uses let, const, and var to declare variables.

// Variable Declaration
let name = "John";
const age = 30;
var isStudent = true;

// Output variables
console.log(name, age, isStudent);

üî§ Data Types

JavaScript has dynamic types. The same variable can hold different data types.

// Primitive Data Types
let string = "Hello World";
let number = 42;
let boolean = true;
let nullValue = null;
let undefinedValue;

// Output types
console.log(typeof string, typeof number);

üìä Arrays

Arrays are used to store multiple values in a single variable.

// Array Declaration
let fruits = ["Apple", "Banana", "Orange"];
let numbers = [1, 2, 3, 4, 5];

// Access array elements
console.log(fruits[0]); // Apple
console.log(numbers.length); // 5

üì¶ Objects

Objects are collections of key-value pairs used to store related data.

// Object Declaration
let person = {
name: "Alice",
age: 25,
isStudent: false,
hobbies: ["reading", "coding"]
};

// Access object properties
console.log(person.name);
console.log(person.hobbies[0]);

‚öôÔ∏è Functions

Functions are blocks of code designed to perform a particular task.

// Function Declaration
function greet(name) {
return "Hello, " + name + "!";
}

// Function Call
let message = greet("Sarah");
console.log(message); // Hello, Sarah!

üîÄ Parameters & Spread/Rest

Functions can accept parameters. Spread and rest operators provide flexibility.

// Rest Parameters
function sum(...numbers) {
return numbers.reduce((acc, val) => acc + val, 0);
}

// Spread Operator
let arr1 = [1, 2, 3];
let arr2 = [...arr1, 4, 5];

console.log(sum(1, 2, 3, 4)); // 10
console.log(arr2); // [1, 2, 3, 4, 5]

‚Ü©Ô∏è Return Statements

The return statement stops function execution and returns a value.

// Function with return
function multiply(a, b) {
return a * b;
}

// Using the returned value
let result = multiply(5, 3);
console.log(result); // 15

// Function without return
function sayHello(name) {
console.log("Hello, " + name);
}

let greeting = sayHello("Tom");
console.log(greeting); // undefined

üîÑ Loops

Loops execute a block of code multiple times.

// For Loop
for (let i = 0; i < 5; i++) {
console.log("Iteration: " + i);
}

// While Loop
let count = 0;
while (count < 3) {
console.log("Count: " + count);
count++;
}

// For...of Loop (ES6+)
let colors = ["red", "green", "blue"];
for (let color of colors) {
console.log(color);
}

üîí Closures

A closure gives you access to an outer function's scope from an inner function.

// Closure Example
function createCounter() {
let count = 0;
return function() {
count++;
return count;
};
}

let counter = createCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

üìû Callbacks

A callback is a function passed as an argument to another function.

// Callback Example
function processUserInput(callback) {
let name = "John Doe";
callback(name);
}

function greeting(name) {
console.log("Hello, " + name);
}

processUserInput(greeting); // Hello, John Doe

// Asynchronous callback
setTimeout(function() {
console.log("This runs after 1 second");
}, 1000);

üß∞ Array Methods

JavaScript provides many built-in methods to work with arrays.

let numbers = [1, 2, 3, 4, 5];

// map() - creates new array
let doubled = numbers.map(num => num * 2);

// filter() - filters elements
let evens = numbers.filter(num => num % 2 === 0);

// reduce() - reduces to single value
let sum = numbers.reduce((acc, num) => acc + num, 0);

console.log(doubled); // [2, 4, 6, 8, 10]
console.log(evens); // [2, 4]
console.log(sum); // 15

‚ú® ES6+ Features

Modern JavaScript (ES6+) introduced many new features.

// Arrow Functions
const add = (a, b) => a + b;

// Template Literals
let name = "Alice";
console.log(`Hello, ${name}!`);

// Destructuring
let [first, second] = [10, 20];
let {age, city} = {age: 30, city: "NYC"};

// Classes
class Person {
constructor(name) {
this.name = name;
}
greet() {
return `Hello, I'm ${this.name}`;
}
}
